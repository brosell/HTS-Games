#pragma warning (disable : 4786 4503 4530)
#include "Label.h"

#include "SDL_gfxPrimitives.h"

void Label::paint(Display* surface)
{
  Rect r = getTranslatedPosition();

  int dx, dy;
  
  switch(m_vertJust)
  {
  case JUST_TOP:
    dy = r.y;
    break;
  case JUST_BOTTOM:
    dy = (r.y+r.h) -getFont()->getFontHeight();
    break;
  case JUST_MIDDLE:
    dy = (r.y+(r.h/2)) -(getFont()->getFontHeight()/2);
    break;
  default:
    lassert(false);
  }

  switch(m_horizJust)
  {
  case JUST_LEFT:
    dx = r.x;
    break;
  case JUST_RIGHT:
    dx = (r.x+r.w) -(getFont()->getTextWidth(m_text));
    break;
  case JUST_CENTER:
    dx = (r.x+(r.w/2)) -(getFont()->getTextWidth(m_text)/2);
    break;
  default:
    lassert(false);
  }

  getFont()->drawLine(surface, m_text, dx, dy);
}
